    <script>

        const mergeBtn = document.getElementById('merge-btn');

        const resetBtn = document.getElementById('reset-btn');

        const tableBody = document.querySelector('#schedule-table tbody');

        

        let selectedCells = [];


        // Single-click to select cells for merging

        tableBody.addEventListener('click', (event) => {

            const cell = event.target;

            // Check that the cell is a schedule cell and not the time column

            if (cell.tagName === 'TD' && !cell.classList.contains('time-cell')) {

                cell.classList.toggle('selected-cell');

                

                // Add or remove the cell from the selectedCells array

                const index = selectedCells.indexOf(cell);

                if (index > -1) {

                    selectedCells.splice(index, 1);

                } else {

                    selectedCells.push(cell);

                }

            }

        });


        // Double-click to edit a cell

        tableBody.addEventListener('dblclick', (event) => {

            const cell = event.target;

            // Check that the cell is a schedule cell and not the time column

            if (cell.tagName === 'TD' && !cell.classList.contains('time-cell')) {

                cell.contentEditable = 'true';

                cell.focus();

            }

        });


        // Blur event to stop editing when the user clicks away

        tableBody.addEventListener('blur', (event) => {

            const cell = event.target;

            if (cell.tagName === 'TD') {

                cell.contentEditable = 'false';

            }

        }, true); // Use capture phase to ensure it works on the target


        mergeBtn.addEventListener('click', () => {

            // Sort selected cells by row and then by column to handle selection order

            selectedCells.sort((a, b) => {

                const rowA = a.closest('tr').rowIndex;

                const rowB = b.closest('tr').rowIndex;

                const colA = a.cellIndex;

                const colB = b.cellIndex;

                if (colA !== colB) return colA - colB;

                return rowA - rowB;

            });

            

            if (selectedCells.length > 1) {

                // Check if all cells are in the same column

                const firstCellColumn = selectedCells[0].cellIndex;

                const allInSameColumn = selectedCells.every(cell => cell.cellIndex === firstCellColumn);


                // Check if cells are contiguous

                const firstRow = selectedCells[0].closest('tr').rowIndex;

                const lastRow = selectedCells[selectedCells.length - 1].closest('tr').rowIndex;

                const isContiguous = (lastRow - firstRow + 1) === selectedCells.length;


                if (allInSameColumn && isContiguous) {

                    const firstCell = selectedCells[0];

                    firstCell.rowSpan = selectedCells.length;


                    // Concatenate the content of the merged cells

                    let mergedContent = '';

                    selectedCells.forEach(cell => {

                        mergedContent += cell.innerText.trim() + ' ';

                    });

                    firstCell.innerText = mergedContent.trim();

                    

                    // Hide the remaining cells

                    for (let i = 1; i < selectedCells.length; i++) {

                        selectedCells[i].style.display = 'none';

                    }

                }

            }


            // Clear the selection regardless of whether a merge occurred

            selectedCells.forEach(cell => cell.classList.remove('selected-cell'));

            selectedCells = [];

        });


        resetBtn.addEventListener('click', () => {

            // A simple way to reset the calendar is to reload the page

            window.location.reload();

        });

    </script>
